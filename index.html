Default Parameters


        function func(sizeArg) {
          var size = sizeArg || 0;
        }
        function func(size = 0) {
          return size;
        }
        console.log(func());
        // 0

Map, WeakMap


        var elMap = new Map();
        var el = document.getElementById('size');
        if (el) {
          elMap.set(el, 3);
        }

        // meanwhile ... somewhere else.
        let size = document.getElementById('size');
        if (elMap.has(size)) {
          let size = elMap.get(size)
          console.log(size);
          // 3
        }

Set, WeakSet


        var items = [1, 1, 2, 3];
        var itemSet = new Set(items);
        var uniqueItems = Array.from(itemSet.entries());
        console.log(uniqueItems);
        // [1, 2, 3]

Promises


        function getItems(options) {
          return new Promise(resolve, reject) {
            var {url} = options;
            $.ajax(url)
              .done(function(resp) {
                resolve(resp)
              })
              .fail(function(e) {
                reject(e);
              });
          }
        }
        Promise.all([
          getItems({url: 'http://firstapi.com'}),
          getItems({url: 'http://secondapi.com'})
        ])
        .then(responses => {
          var [firstResponse, secondResponse] = responses;
          console.log(firstResponse);
          console.log(secondResponse);
        })
        .catch(e => {console.error(e)});
